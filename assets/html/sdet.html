// Selenium WebDriver — Java + TestNG example (National Bank SDET stack)
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.annotations.*;

public class LoginPageTest {

  private WebDriver driver;
  private LoginPage loginPage;

  @BeforeClass
  public void setUp() {
    driver = new ChromeDriver();
    driver.manage().window().maximize();
    loginPage = new LoginPage(driver);
  }

  @Test(description = "Valid login navigates to dashboard")
  public void validLoginTest() {
    loginPage.navigateTo("https://app.example.com/login");
    loginPage.enterCredentials("user@bank.ca", "SecureP@ss1");
    loginPage.clickLogin();

    // Explicit wait — preferred over Thread.sleep
    WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(10));
    wait.until(ExpectedConditions.urlContains("/dashboard"));
    Assert.assertTrue(driver.getCurrentUrl().contains("/dashboard"));
  }

  @Test(description = "Invalid credentials show error message")
  public void invalidLoginTest() {
    loginPage.navigateTo("https://app.example.com/login");
    loginPage.enterCredentials("wrong@bank.ca", "badpass");
    loginPage.clickLogin();

    WebElement error = driver.findElement(By.cssSelector(".error-msg"));
    Assert.assertTrue(error.isDisplayed(), "Error message should appear");
    Assert.assertEquals(error.getText(), "Invalid email or password.");
  }

  @AfterClass
  public void tearDown() { driver.quit(); }
}
Page Object Model pattern ⏱️ Explicit WebDriverWait (no Thread.sleep) ☕ Java + TestNG 🏦 National Bank tech stack 
// Playwright — TypeScript, auto-wait, parallel execution
import { test, expect } from '@playwright/test';

test.describe('API + UI Integration Tests', () => {

  test('intercepts API call and validates UI response', async ({ page }) => {
    // Intercept the backend API and return a mock response
    await page.route('**/api/dashboard', async route => {
      await route.fulfill({
        status: 200,
        body: JSON.stringify({ totalTests: 142, passRate: '96.5%' })
      });
    });

    await page.goto('/dashboard');

    // Playwright auto-waits — no manual waits needed
    await expect(page.locator('#total-tests')).toHaveText('142');
    await expect(page.locator('#pass-rate')).toHaveText('96.5%');
  });

  test('visual regression — screenshot comparison', async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // Full-page pixel-diff snapshot test
    await expect(page).toHaveScreenshot('homepage.png', {
      fullPage: true,
      threshold: 0.02  // 2% pixel diff tolerance
    });
  });
});
